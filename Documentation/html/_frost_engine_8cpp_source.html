156 void Application::PhysUpdate()
157 {
158 std::map < frost::ECS::BoxCollider*, std::vector < frost::ECS::CollisionData >> Hits;
159
160 for (auto* object : ObjectsToUpdate)
161 {
162 auto* boxCollider = object->GetComponent<frost::ECS::BoxCollider>();
163
164 if (boxCollider)
165 {
166 for (auto* otherObject : ObjectsToUpdate)
167 {
168 auto* otherBoxCollider = otherObject->GetComponent<frost::ECS::BoxCollider>();
169
170 if (otherBoxCollider && otherObject != object)
171 {
172 auto CollisionData = boxCollider->IsColliding(*otherBoxCollider);
173 if (CollisionData.isColliding)
174 {
175 Hits[boxCollider].push_back(CollisionData);
176 }
177 }
178 }
179 }
180 }
181
182 // foreach element on Hits map, iterate on the vector find the largets value on each field of CollisionData
183 for (auto& [boxCollider, collisions] : Hits)
184 {
185 auto* transform = boxCollider->GetParentObject().GetComponent<frost::ECS::Transform>();
186
187 if (transform && !transform->isMovingEntity)
188 continue;
189
190 frost::ECS::CollisionData largestCollision = { false, 0, 0, 0, 0 };
191
192 for (auto& collision : collisions)
193 {
194 boxCollider->GetParentObject().OnCollisionEnter(&collision);
195
196 if (collision.top > largestCollision.top)
197 largestCollision.top = collision.top;
198
199 if (collision.bottom < largestCollision.bottom)
200 largestCollision.bottom = collision.bottom;
201
202 if (collision.left < largestCollision.left)
203 largestCollision.left = collision.left;
204
205 if (collision.right > largestCollision.right)
206 largestCollision.right = collision.right;
207 }
208
209 // Apply the collision
210 if (transform)
211 {
212 float absTop = abs(largestCollision.top);
213 float absBottom = abs(largestCollision.bottom);
214 float absLeft = abs(largestCollision.left);
215 float absRight = abs(largestCollision.right);
216
217 // find the smallest offset
218 if (absTop < absBottom && absTop < absLeft && absTop < absRight && absTop != 0)
219 transform->position.y -= abs(largestCollision.top);
220 else if (absBottom < absTop && absBottom < absLeft && absBottom < absRight && absBottom != 0)
221 transform->position.y += abs(largestCollision.bottom);
222 else if (absLeft < absTop && absLeft < absBottom && absLeft < absRight && absLeft != 0)
223 transform->position.x += abs(largestCollision.left);
224 else if (absRight < absTop && absRight < absBottom && absRight < absLeft && absRight != 0)
225 transform->position.x -= abs(largestCollision.right);
226 }
227 }
228 }
