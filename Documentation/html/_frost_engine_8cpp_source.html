149 void Application::PhysUpdate()
150 {
151 std::map < frost::ECS::BoxCollider*, std::vector < frost::ECS::CollisionData >> Hits;
152
153 for (auto* object : ObjectsToUpdate)
154 {
155 auto* boxCollider = object->GetComponent<frost::ECS::BoxCollider>();
156
157 if (boxCollider)
158 {
159 for (auto* otherObject : ObjectsToUpdate)
160 {
161 auto* otherBoxCollider = otherObject->GetComponent<frost::ECS::BoxCollider>();
162
163 if (otherBoxCollider && otherObject != object)
164 {
165 auto CollisionData = boxCollider->IsColliding(*otherBoxCollider);
166 if (CollisionData.isColliding)
167 {
168 Hits[boxCollider].push_back(CollisionData);
169 }
170 }
171 }
172 }
173 }
174
175 // foreach element on Hits map, iterate on the vector find the largets value on each field of CollisionData
176 for (auto& [boxCollider, collisions] : Hits)
177 {
178 auto* transform = boxCollider->GetParentObject().GetComponent<frost::ECS::Transform>();
179
180 if (transform && !transform->isMovingEntity)
181 continue;
182
183 frost::ECS::CollisionData largestCollision = { false, 0, 0, 0, 0 };
184
185 for (auto& collision : collisions)
186 {
187 boxCollider->GetParentObject().OnCollisionEnter(&collision);
188
189 if (collision.top > largestCollision.top)
190 largestCollision.top = collision.top;
191
192 if (collision.bottom < largestCollision.bottom)
193 largestCollision.bottom = collision.bottom;
194
195 if (collision.left < largestCollision.left)
196 largestCollision.left = collision.left;
197
198 if (collision.right > largestCollision.right)
199 largestCollision.right = collision.right;
200 }
201
202 // Apply the collision
203 if (transform)
204 {
205 float absTop = abs(largestCollision.top);
206 float absBottom = abs(largestCollision.bottom);
207 float absLeft = abs(largestCollision.left);
208 float absRight = abs(largestCollision.right);
209
210 // find the smallest offset
211 if (absTop < absBottom && absTop < absLeft && absTop < absRight && absTop != 0)
212 transform->position.y -= abs(largestCollision.top);
213 else if (absBottom < absTop && absBottom < absLeft && absBottom < absRight && absBottom != 0)
214 transform->position.y += abs(largestCollision.bottom);
215 else if (absLeft < absTop && absLeft < absBottom && absLeft < absRight && absLeft != 0)
216 transform->position.x += abs(largestCollision.left);
217 else if (absRight < absTop && absRight < absBottom && absRight < absLeft && absRight != 0)
218 transform->position.x -= abs(largestCollision.right);
219 }
220 }
221 }
